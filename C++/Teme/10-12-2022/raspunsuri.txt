6.#include <iostream>
#include <fstream>

using namespace std;

int main(){
    int n, np, ni, cop;
    ifstream f("date.in");
    while(f >> n){
        np = 0;
        ni = 0;
        cop = n;
        while(cop!=0){
            if(cop%10%2==0){
                np = np + 1;
            } else {
                ni = ni + 1;
            }
            cop = cop/10;
        }
        if(np==ni){
            cout << n << endl;
        }
    }

    return 0;
}

// date.in > 32 42 123124 2363

7.#include <iostream>
#include <fstream>

using namespace std;

int main(){
    int n, i, min=9, max=0, x;
    ifstream f("numere.txt");
    f >> n;
    for(i=1;i<=n;i++){
        f >> x;
        if(min>x){
            min=x;
        }
        if(max<x){
            max=x;
        }
    }
    cout << max << " " << min;
    return 0;
}
